var Snapdragon = require('snapdragon');
var captureSet = require('snapdragon-capture-set');
var parser = new Snapdragon.Parser()
  .use(captureSet())
  .captureSet('brace', /^\{/, /^\}/) 
  .set('text', function() {
    var pos = this.position();
    var m = this.match(/^[^{}]/);
    if (!m) return;
    return pos({
      type: 'text',
      val: m[0]
    });
  });

var ast = parser.parse('a{b,{c,d},e}f');
console.log(ast.nodes[2]);
